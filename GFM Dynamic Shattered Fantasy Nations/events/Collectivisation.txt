#Collectivisation - start
country_event = {
    id = 1512058
    title = "Collectivisation in $COUNTRY$"
    desc = EVTDESC1512058
    picture = "collectivisation"

    trigger = {

        has_country_flag = collectivise_farms

        any_owned_province = {
                OR = {
                    trade_goods = fruit
                    trade_goods = grain
                    trade_goods = tea
                    trade_goods = tobacco
                    trade_goods = coffee
                    trade_goods = cotton
                }
                NOT = { has_province_modifier = collectivisation }

        }
    }

    option = {
        name = "This will increase crop yield!"

        any_owned = {
            limit = {
                OR = {
                    trade_goods = fruit
                    trade_goods = grain
                    trade_goods = tea
                    trade_goods = tobacco
                    trade_goods = coffee
                    trade_goods = cotton
                }
                NOT = { has_province_modifier = collectivisation }
            }
            add_province_modifier = { name = collectivisation duration = -1 }
        }

    }

}
