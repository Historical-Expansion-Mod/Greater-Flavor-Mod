# Inorganic chemestry and Prophylaxis against Malaria makes the Amazon less hellish
country_event = {
    id = 47115
    title = "Taming the Green Hell"
    desc = "green_hell_mid_event_desc"
    picture = "green_hell"

    fire_only_once = yes
    mean_time_to_happen = { days = 1 }

    trigger = {
        has_global_flag = green_hell_start_flag
        OR = { capital_scope = { continent = south_america } tag = UPB }
        OR = {
            AND = {
                organic_chemistry = 1
                invention = prophylaxis_against_malaria
            }
            AND = {
                year = 1865
                ai = yes
            }
        }
    }

    option = {
        name = "Huzzah!"
        prestige = 5
        any_owned = {
            limit = { has_province_modifier = green_hell_start }
            remove_province_modifier = green_hell_start
            add_province_modifier = { name = green_hell_mid duration = -1 }
        }
        set_global_flag = green_hell_mid_flag
        clr_global_flag = green_hell_start_flag
    }
}
