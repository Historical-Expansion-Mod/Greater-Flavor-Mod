# Battle of Oncativo - fires for LPL - sets in motion the Argentine civil war

	id = 16051925
	title = "The Battle of Oncativo"
	desc = "BATTLEOFONCATIVO"

    fire_only_once = yes

	trigger = {
		tag = LPL
		ARC = { exists = no }
		year = 1830
		month = 2
		NOT = { year = 1832 }
	}
	
    mean_time_to_happen = { days = 1 }

	option = {
		name = "We must unleash our dogs of war."
		any_owned = {
			limit = {
				NOT = {
					is_core = CRT
					is_core = ENT
					is_core = SBA

					province_id = 2363
					province_id = 4069
					province_id = 2131
				}
			}
			any_pop = {
				militancy = 3
				consciousness = 3
			}
		}
		set_country_flag = oncativo
	}
}

# Fraile Muerto - fires for ARC - just before the Argentine Civil war - caused by event [Id = 16051927] in "events/1830_GFM_ARG.txt"
country_event = {
	id = 16051928
	title = "The Battle of Fraile Muerto"
	desc = "UNITARYLEAGUEWAR"
	picture = "fraile_muerto"
    fire_only_once = yes

	trigger = {
		year = 1831
		tag = ARC
		has_country_flag = pacto_federal
		month = 5
		NOT = { year = 1833 }
	}
	
    mean_time_to_happen = { days = 1 }

	option = {
		name = "Federalism shall win!"
		random_owned = {
			limit = { owner = { ai = yes ARG = { ai = yes } } }
			owner = {
				ARC = { add_country_modifier = { name = small_country_draft_big duration = 730 } }
				ENT = { add_country_modifier = { name = small_country_draft_big duration = 730 } }
				CRT = { add_country_modifier = { name = small_country_draft_big duration = 730 } }
				add_country_modifier = {
					name = army_power_boost
					duration = 730
				}
			}
		}
		any_country = {
			limit = { alliance_with = ARC }
			leave_alliance = ARC
		}
		relation = { who = ENT value = 100 }
		relation = { who = CRT value = 100 }
		create_alliance = ENT
		create_alliance = CRT
		release_vassal = ARG
		ARG = { country_event = 16051929 }
		set_global_flag = argentine_civil_war
	}
}

# Argentine Civil War win event for ARC - fires for ARC if ARG has been defeated
country_event = {
	id = 16051930
	title = "José María Paz Captured"
	desc = "PAZCAPTURED"
	picture = "paz_captured"
    fire_only_once = yes

    trigger = {
		has_global_flag = argentine_civil_war
		tag = ARC
		has_global_flag = eighteenthirty_startdate
		OR = {
			AND = {
				NOT = { war_with = ARG }
				ARG = { exists = yes }
			}
			AND = {
				ARG = { exists = no }
				war = no
			}
		}
	}
	
    mean_time_to_happen = { days = 1 }

	option = {
		name = "His movement ends with his capture."
		ARC = { all_core = { remove_core = ARG } }
		government = presidential_dictatorship
		add_country_modifier = {
			name = america_new_world
			duration = 10950
		}
		random_owned = {
			limit = {
				owner = { ARG = { exists = yes } }
			}
			owner = { ARC = { inherit = ARG } }
		}
		clr_global_flag = argentine_civil_war
	}
}
