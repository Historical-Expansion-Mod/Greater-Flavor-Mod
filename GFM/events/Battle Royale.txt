country_event = {
	id = 93759275
	title = "Battle Royale"
	desc = "The entire world is engulfed in an eternal war."
	picture = "crimean_war_defeat"
	trigger = {
		has_global_flag = battle_royale
		is_vassal = no
		OR = {
			war = no
			AND = {
				any_land_province = {
					empty = no
					is_coastal = yes
					NOT = { owned_by = THIS }
					owner = {
						NOT = { war_with = THIS }
						is_vassal = no
					}
				}
				NOT = {
					any_neighbor_country = {
						exists = yes
						neighbour = THIS
					}
				}
			}
			AND = {
				any_neighbor_country = {
					is_vassal = no
					exists = yes
					neighbour = THIS
					NOT = { war_with = THIS }
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = "War, war never changes."
		any_neighbor_country = {
			limit = {
				is_vassal = no
				exists = yes
				neighbour = THIS
				NOT = { war_with = THIS }
			}
			war = {
				target = THIS
				attacker_goal = { casus_belli = conquest_any }
				defender_goal = { casus_belli = conquest_any }
			}
		}
		random_owned = {
			limit = {
				owner = {
					NOT = {
						any_neighbor_country = {
							exists = yes
							neighbour = THIS
						}
					}
				}
			}
			owner = {
				any_land_province = {
					limit = {
						empty = no
						is_coastal = yes
						NOT = { owned_by = THIS }
						owner = {
							NOT = { war_with = THIS }
							is_vassal = no
						}
					}
					owner = {
						war = {
							target = THIS
							attacker_goal = { casus_belli = conquest_any }
							defender_goal = { casus_belli = conquest_any }
						}
					}
				}
			}
		}
	}
}

country_event = {
	id = 93759279
	title = "Battle Royale Setup"
	desc = "With global war looming on the horizon, it is time to say goodbye to our alliances."
	picture = "crimean_war_defeat"
	is_triggered_only = yes

	option = {
		name = "Huzzah!"
		any_country = {
			relation = {
				who = THIS
				value = -400
			}
			diplomatic_influence = {
				who = THIS
				value = -400
			}
		}
		any_country = {
			limit = {
				war_with = THIS
			}
			end_war = THIS
		}
		any_country = {
			limit = {
				alliance_with = THIS
			}
			leave_alliance = THIS
		}
		random_owned = {
			limit = {
				owner = {
					is_vassal = yes
				}
			}
			owner = {
				overlord = { release_vassal = THIS }
			}
		}
	}
}
