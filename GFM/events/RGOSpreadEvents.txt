# Tractors
country_event = {
    id = 29900
    title = "Tractors"
    desc = "EVTDESC29900"
	picture = "tractors"

    trigger = {
		invention = tractors
        any_owned_province = {
			NOT = { has_province_modifier = local_tractors }
			NOT = { state_scope = { any_owned_province = { NOT = { controlled_by = THIS } } } }
			NOT = { state_scope = { is_slave = yes } }
			NOT = { state_scope = { has_pop_type = serfs } }
			is_overseas = no
			is_colonial = no
			state_scope = { literacy = 0.7 }
			OR = {
				terrain = plains
				terrain = steppe
				terrain = savanna
				terrain = forest
				terrain = hills
				terrain = dryhills
				terrain = farmlands
				terrain = floodplain_farmland
				terrain = mountain
			}
			OR = {
				trade_goods = tobacco
				trade_goods = tea
				trade_goods = coffee
				trade_goods = grain
				trade_goods = opium
				trade_goods = cotton
				trade_goods = fruit
			}
		}
		NOT = { has_country_flag = tractored }
    }

    allow_multiple_instances = no

    mean_time_to_happen = {
        months = 1
    }

    option = {
        name = "EVTOPTA29900"
		set_country_flag = tractored
		any_owned = {
			limit = {
				state_scope = { is_slave = no }
				NOT = { state_scope = { has_pop_type = serfs } }
				is_overseas = no
				is_colonial = no
				NOT = { state_scope = { any_owned_province = { NOT = { controlled_by = THIS } } } }
				NOT = { has_province_modifier = local_tractors }
				state_scope = { literacy = 0.7 }
				OR = {
					terrain = plains
					terrain = steppe
					terrain = savanna
					terrain = forest
					terrain = hills
					terrain = dryhills
					terrain = farmlands
					terrain = floodplain_farmland
					terrain = mountain
				}
				OR = {
					trade_goods = tobacco
					trade_goods = tea
					trade_goods = coffee
					trade_goods = grain
					trade_goods = opium
					trade_goods = cotton
					trade_goods = fruit
				}
			}
            add_province_modifier = { name = local_tractors duration = -1 }
        }
    }
}
