# Event to set Ranked List
country_event = { # To initiate the count
    id = 81000
	major = yes
    title = "Configure Countries to Rank"
    desc = "EVTDESC70999"
    picture = "treaty"
    is_triggered_only = yes

    option = {
        name = "EVTOPTA70999"
		any_country = { # Clean up previous use
			set_variable = {
				which = accending_group_rank
				value = 0
			}
			set_variable = {
				which = decending_group_rank
				value = 0
			}
			clr_country_flag = rank_set
		}
		any_country = { 
			limit = {
				has_country_flag = rank
			}
			CLN = { country_event = 81001 }
		}
		CLN = { country_event = 81003 }
    }	
}

country_event = { # Spammed event to set each target with a variable with total of countries scoped
    id = 81001
	major = yes
    title = "EVTNAME70999"
    desc = "EVTDESC70999"
    picture = "treaty"
	allow_multiple_instances = yes
    is_triggered_only = yes

    option = {
        name = "EVTOPTA70999"
		any_country = { 
			limit = {
				has_country_flag = rank

    }
}
