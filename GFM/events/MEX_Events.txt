#The Goliad Massacre occurs when Mexico occupies San Antonio during the war with Texas
country_event = {
	id = 4206989
	title = "The Goliad Massacre"
	desc = "EVTDESC4206989"
	picture = "goliad_massacre"
	fire_only_once = yes

	trigger = {
		tag = MEX
		war_with = TEX
		year = 1836
		has_country_modifier = santa_ana_rule
		NOT = {
			has_country_flag = goliad_massacre
			year = 1838
		}
		137 = { controlled_by = MEX }
	}

	mean_time_to_happen = { days = 15 }

	option = {
		name = "They got what they deserve."
		set_country_flag = goliad_massacre
		any_pop = {
			limit = {
				has_pop_culture = dixie
			}
			militancy = 3
			consciousness = 4
		}
		TEX = {
			any_pop = {
				limit = {
					type = soldiers
				}
				reduce_pop = 0.9
			}
			war_exhaustion = 5
		}
	}
}
