country_event = { # Prime Colonial Territory - forces historical colonies/borders
    id = 20541
    title = "Assign Colonial Land"
    is_triggered_only = yes
	
	immediate = {
		# Colony of Morocoo																																						# Name
		any_land_province = {
			limit = {
				OR = {																																							# Provinces of the Colonial Region
					provid = 1687
					provid = 1686
					provid = 3865
					provid = 1688
				}
				CLN = { NOT = { has_global_flag = dismantlement_colony_buffer } }
			}
			add_province_modifier = { name = dismantlement_colony duration = -1 }
			set_global_flag = dismantlement_colony_buffer
			CLN = { 
				set_global_flag = dismantlement_colony_buffer 
				slot_10 = { secede_province = THIS state_scope = { change_region_name = "Morocco" } secede_province = QQQ }														# State Name for Event
			}
			# Colonies																																							# Colonies to merge into
			MOR = { set_country_flag = dismantlement_colony_target }																											
		}
		CLN = { limit = { has_global_flag = dismantlement_colony_total_buffer } change_variable = { which = dismantlement_colony_total value = 1 } }
	}
	
	option = {
        name = "Colonial Flags set"
    }
}
