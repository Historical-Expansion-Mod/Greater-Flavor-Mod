country_event = { # For Gallaecian countries
    id = 9000091
    title = "EVTNAME9000091"
    desc = "EVTDESC9000091"
    picture = "iberian_hegemony"
    is_triggered_only = yes
        
    option = {
        name = "The Spanish Lion has been defeated."
        any_owned = { 
            limit = { is_core = CAT }
            secede_province = CAT
        }
        CAT = { create_alliance = THIS }
        any_owned = { 
            limit = { is_core = BSQ }
            secede_province = BSQ
        }
        # BASQUE BLOCK
        BSQ = { create_alliance = THIS }
        # CASTILE BLOCK
        random_owned = {
            limit = { owner = { OR = { government = absolute_monarchy AND = { tag = SPC government = absolute_empire } } } }
            owner = { change_tag_no_core_switch = SPA government = absolute_empire }
        }
        random_owned = { 
            limit = { owner = { OR = { government = prussian_constitutionalism AND = { tag = SPC government = semi_constitutional_empire } } } }
            owner = { change_tag_no_core_switch = SPA government = semi_constitutional_empire }
        }
        random_owned = { 
            limit = { owner = { OR = { government = hms_government AND = { tag = SPC government = constitutional_empire } } } }
            owner = { change_tag_no_core_switch = SPA government = constitutional_empire }
        }
        any_country = { limit = { has_country_flag = gallaecian_victory } country_event = { id = 9000092 } }
    }
}
