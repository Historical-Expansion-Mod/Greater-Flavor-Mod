            limit = { owner = { NOT = { has_country_flag = end_of_the_confederation } } }
            owner = {
				PEU = { all_core = { add_core = PBC } }
                SPU = { all_core = { add_core = PBC remove_core = SPU } }
			}
        }
        BOL = { all_core = { add_core = PBC } }
        any_country = {
            limit = {
                OR = {
                	tag = SPU

                	in_sphere = PBC
                	vassal_of = PBC

            }

        }
        random_country = {
            limit = { tag = SPU exists = no }
            SPU = { all_core = { remove_core = SPU } }
        }
		ai_chance = { factor = 1 }
    }
}
