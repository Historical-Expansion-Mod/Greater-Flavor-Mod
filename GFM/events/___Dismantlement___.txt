				exists = yes
			}
			set_country_flag = dismantlement_independent_core
		}

		# Designate Vassal Targets
		any_country = {
			limit = {
				exists = yes
				OR = { overlord = { has_country_flag = dismantlement_targets } has_country_flag = dismantlement_targets }
				NOT = { has_country_flag = dismantlement_target }
			}
			set_country_flag = dismantlement_rewriteable_vassal
		}
		
		# Designate Non-Existing Targets
		any_country = {
			limit = {
				exists = no
				any_core = { owner = { OR = { overlord = { has_country_flag = dismantlement_targets } has_country_flag = dismantlement_targets } } }
				NOT = { has_country_flag = dismantlement_target AND = { has_country_flag = colonial_exception civilised = no } }
			}
			set_country_flag = dismantlement_rewriteable_core
			set_country_flag = dismantlement_targets
		}
		set_global_flag = dismantlement_active 

	}
	
	option = {
        name = "Trigger Dismantlement Bounce Event"
