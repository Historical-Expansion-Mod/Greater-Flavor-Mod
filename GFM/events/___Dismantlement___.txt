country_event = { # Scope the cores to release
    id = 20420
    title = "Dismantlement Assignment"
    is_triggered_only = yes
	
	immediate = {
		# Will need to fire Colonial Config Event here
		country_event = 20520
		# Give cores to the Leader
		any_land_province = {
			limit = { owner = {	has_country_flag = dismantlement_targets } any_core = { has_country_flag = dismantlement_leader }  }
			secede_province = THIS
			add_core = UTI
		}
		release = UTI
		UTI = { annex_to = UTI all_core = { remove_core = UTI } }
		random_country = {
			limit = {
				has_country_flag = dismantlement_leader
			}
			inherit = UTI
		}
		
		# Designate Independent Targets
		any_country = {
			limit = {
				any_core = { owner = { has_country_flag = dismantlement_targets } }
				NOT = { overlord = { has_country_flag = dismantlement_targets } has_country_flag = dismantlement_targets }
				exists = yes
			}
			set_country_flag = dismantlement_independent_core
		}

		# Designate Vassal Targets
		any_country = {
			limit = {
				exists = yes
				OR = { overlord = { has_country_flag = dismantlement_targets } has_country_flag = dismantlement_targets }
				NOT = { has_country_flag = dismantlement_target }
			}
			set_country_flag = dismantlement_rewriteable_vassal
		}
		
		# Designate Non-Existing Targets
		any_country = {
			limit = {
				exists = no
				any_core = { owner = { OR = { overlord = { has_country_flag = dismantlement_targets } has_country_flag = dismantlement_targets } } }
				NOT = { has_country_flag = dismantlement_target AND = { has_country_flag = colonial_exception civilised = no } }
			}
			set_country_flag = dismantlement_rewriteable_core
			set_country_flag = dismantlement_targets
		}
		set_global_flag = dismantlement_active 

	}
	
	option = {
        name = "Trigger Dismantlement Bounce Event"
    }
}
