# Sets Andrew Jackson as President in 1830
country_event = {
	
	id = 1010100
	
	title = "Dynamic Leaders: Setting Andrew Jackson as president" 
	desc = "Dynamic Leaders: Setting Andrew Jackson as president"
	picture = "danishgovernment"
	
	fire_only_once = yes 
	
	trigger = { 
		
		tag = CLN
		
		OR = {
			
			AND = {
				year = 1830
				NOT = { year = 1833 }
			}
			AND = {
				year = 1836
				NOT = { year = 1837 }
			}
			
		}
		
		USA = {
			
			OR = {
				government = democracy
				has_country_flag = Government_Democracy
			}
			
			NOT = {
				has_country_modifier = andrew_jackson
				has_country_modifier = henry_clay
			}
			
		}
		
	}
	
	mean_time_to_happen = { days = -1 }
	
	option = {
		
		name = "Setting..."
		
		USA = {
			
			add_country_modifier = { 
				name = andrew_jackson
				duration = -1
			}
			
		}
		
	}
	
}

	id = 1010127
	
	title = "Dynamic Leaders: Setting Abraham Lincoln as president" 
	desc = "Dynamic Leaders: Setting Abraham Lincoln as president"
	picture = "danishgovernment"
	
	trigger = { 
		tag = CLN
		
		year = 1860
		NOT = { year = 1864 }
		
		USA = {
			
			has_country_flag = liberal_election_win
			
			OR = {
				government = democracy
				has_country_flag = Government_Democracy
			}
			
		}
		
	}
	
	mean_time_to_happen = { days = -1 }
	
	immediate = {
		clr_country_flag = liberal_election_win
	}
	
	option = { 
		name = "Setting..."
		
		USA = {
			
			remove_country_modifier = james_buchanan 
			remove_country_modifier = john_charles_fremont
			remove_country_modifier = john_cabell_breckinridge
			
			add_country_modifier = { 
				name = abraham_lincoln
				duration = -1
			}
			
		}
		
	}
	
}

    }
