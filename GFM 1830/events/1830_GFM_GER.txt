	title = "The Hambacher Festival"
	desc = "HAMBACHFEST1"
	picture = "hambach_festival"
	major = yes

	fire_only_once = yes
	mean_time_to_happen = { months = 6 }

	trigger = {
		tag = BAV
		year = 1832
		owns = 571
	}

	option = {
	    name = "EVTOPTA127847"
	    any_pop = {
			limit = {
				OR = {
					has_pop_culture = north_german
					has_pop_culture = south_german
				}
			}
			consciousness = 4
		}
		any_country = {
			limit = {
				OR = {
					primary_culture = north_german
					primary_culture = south_german
				}
				NOT = {
					OR = {
						tag = SWI
						tag = LIE
						tag = BAN
						tag = TST
					}
				}
			}
			country_event = {id = 127848 days = 1 }
		}
		ai_chance = { factor = 90 }
	}

	option = {
	    name = "EVTOPTB127847"
		any_pop = {
			limit = {
				OR = {
					has_pop_culture = north_german
					has_pop_culture = south_german
				}
			}
			consciousness = -1
			militancy = 2
		}
		ai_chance = {
			factor = 10
        }
    }
}
